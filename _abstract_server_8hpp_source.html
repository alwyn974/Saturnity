83 virtual void sendTo(int id, AbstractPacket &packet)
84 {
85 ByteBuffer buffer;
86 const uint16_t packetId = this->packetRegistry->getPacketId(packet);
87 buffer.writeUShort(packetId);
88 buffer.writeUShort(0);
89 packet.toBytes(buffer);
90 auto size = static_cast<uint16_t>(buffer.writerIndex());
91 buffer.setWriterIndex(sizeof(uint16_t)); // Skip id, for rewrite size
92 buffer.writeUShort(size - sizeof(uint16_t) * 2); // Skip packet id and size
93 buffer.setWriterIndex(size); // Restore writer index
94 this->sendTo(id, buffer);
95 }
