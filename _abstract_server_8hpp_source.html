125 virtual void sendTo(int id, AbstractPacket &packet)
126 {
127 ByteBuffer buffer;
128 const uint16_t packetId = this->packetRegistry->getPacketId(packet);
129 buffer.writeUShort(packetId);
130 buffer.writeUShort(0);
131 packet.toBytes(buffer);
132 auto size = static_cast<uint16_t>(buffer.writerIndex());
133 buffer.setWriterIndex(sizeof(uint16_t)); // Skip id, for rewrite size
134 buffer.writeUShort(size - (sizeof(uint16_t) * 2)); // Write the packet body size
135 buffer.setWriterIndex(size); // Restore writer index
136 this->sendTo(id, buffer);
137 }

161 template<typename T, typename = std::enable_if_t<std::is_base_of<AbstractPacket, T>::value, T>>
162 void registerHandler(std::function<void(std::shared_ptr<ConnectionToClient> &server, T &packet)> handler)
163 {
164 const uint16_t id = this->packetRegistry->getPacketId<T>();
165 if (this->packetHandlers.contains(id))
166 throw ex::PacketHandlerAlreadyExistsException(spdlog::fmt_lib::format("A handler already exists for packet id: {}", id));
167
168 this->packetHandlers[id] = [handler](std::shared_ptr<ConnectionToClient> &server, ByteBuffer &buffer) {
169 T packet;
170 packet.fromBytes(buffer);
171 handler(server, packet);
172 };
173 }

185 this->packetHandlers.erase(id);
