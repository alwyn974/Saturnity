128 virtual void send(AbstractPacket &packet)
129 {
130 if (!this->isConnected()) throw ClientNotConnectedException("The client is not connected to the server.");
131 ByteBuffer buffer;
132 const uint16_t id = this->packetRegistry->getPacketId(packet);
133 buffer.writeUShort(id);
134 buffer.writeUShort(0);
135 packet.toBytes(buffer);
136 auto size = static_cast<uint16_t>(buffer.writerIndex());
137 buffer.setWriterIndex(sizeof(uint16_t)); // Skip id, for rewrite size
138 buffer.writeUShort(size - (sizeof(uint16_t) * 2)); // Write the packet body size
139 buffer.setWriterIndex(size); // Restore writer index
140 this->send(buffer);
141 }

165 template<typename T, typename = std::enable_if_t<std::is_base_of<AbstractPacket, T>::value, T>>
166 void registerHandler(std::function<void(std::shared_ptr<ConnectionToServer> &server, T &packet)> handler)
167 {
168 const uint16_t id = this->packetRegistry->getPacketId<T>();
169 if (this->packetHandlers.contains(id))
170 throw ex::PacketHandlerAlreadyExistsException(spdlog::fmt_lib::format("A handler already exists for packet id: {}", id));
171
172 this->packetHandlers[id] = [handler](std::shared_ptr<ConnectionToServer> &server, ByteBuffer &buffer) {
173 T packet;
174 packet.fromBytes(buffer);
175 handler(server, packet);
176 };
177 }

184 template<typename T, typename = std::enable_if_t<std::is_base_of<AbstractPacket, T>::value, T>>
185 void unregisterHandler()
186 {
187 const uint16_t id = this->packetRegistry->getPacketId<T>();
188 if (!this->packetHandlers.contains(id)) throw ex::PacketHandlerMissingException(spdlog::fmt_lib::format("No handler exists for packet id: {}", id));
189 this->packetHandlers.erase(id);
190 }
