90 virtual void send(AbstractPacket &packet)
91 {
92 if (!this->isConnected()) throw ClientNotConnectedException("The client is not connected to the server.");
93 ByteBuffer buffer;
94 const uint16_t id = this->packetRegistry->getPacketId(packet);
95 buffer.writeUShort(id);
96 buffer.writeUShort(0);
97 packet.toBytes(buffer);
98 auto size = static_cast<uint16_t>(buffer.writerIndex());
99 buffer.setWriterIndex(sizeof(uint16_t)); // Skip id, for rewrite size
100 buffer.writeUShort(size - sizeof(uint16_t) * 2); // Skip packet id and size
101 buffer.setWriterIndex(size); // Restore writer index
102 this->send(buffer);
103 }

126 void registerHandler(std::function<void(std::shared_ptr<ConnectionToServer> &server, T &packet)> handler)
127 {
128 const uint16_t id = this->packetRegistry->getPacketId<T>();
129 if (this->packetHandlers.contains(id))
130 throw PacketHandlerAlreadyExistsException(spdlog::fmt_lib::format("A handler already exists for packet id: {}", id));
131
132 this->packetHandlers[id] = [handler](std::shared_ptr<ConnectionToServer> &server, ByteBuffer &buffer) {
133 T packet;
134 packet.fromBytes(buffer);
135 handler(server, packet);
136 };
