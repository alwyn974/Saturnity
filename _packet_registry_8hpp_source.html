94 void registerPacket(std::uint16_t id)
95 {
96 if (!std::is_default_constructible<T>())
97 throw PacketMissingEmptyConstructorException(spdlog::fmt_lib::format("Packet: {} is missing an empty constructor", typeid(T).name()));
98 if (this->_packetFactories.contains(id))
99 throw PacketAlreadyRegisteredException(spdlog::fmt_lib::format("Packet with id: {} is already registered", id));
100 auto typeIndex = std::type_index(typeid(T)); // NOLINT
101 if (this->_packetRegistry.contains(typeIndex))
102 throw PacketAlreadyRegisteredException(spdlog::fmt_lib::format("Packet with type: {} is already registered", typeIndex.name()));
103 this->_packetRegistry[typeIndex] = id;
104 this->_packetFactories[id] = [](ByteBuffer &buffer) {
105 std::unique_ptr<AbstractPacket> packet = std::make_unique<T>();
106 packet->fromBytes(buffer);
107 return packet;
108 };
109 }
